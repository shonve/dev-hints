var constantize = (obj) => {
  Object.freeze(obj);
  Object.keys(obj).forEach( (key, i) => {
    if ( typeof obj[key] === 'object' ) {
      constantize( obj[key] );
    }
  });
};

  • 去除字符串内左侧的空格：str = str.replace(/^\s*/,””);
  • 去除字符串内右侧的空格：str = str.replace(/(\s*$)/g,””);
